4. Brincando com Números

// Jogo de Adivinhação
function jogarAdivinhar() {
    const numeroSecreto = Math.floor(Math.random() * 20) + 1; // Alterado para números de 1 a 20
    let tentativas = 0;
    let acertou = false;
    
    while (!acertou && tentativas < 5) { // Reduzido para 5 tentativas
        let palpite = prompt("Adivinhe o número (entre 1 e 20):"); // Alterado texto
        tentativas++;
        
        if (palpite == numeroSecreto) {
            acertou = true;
            if (tentativas < melhorPontuacao) {
                melhorPontuacao = tentativas;
                document.getElementById('melhorPontuacao').textContent = melhorPontuacao;
            }
            alert(`Parabéns! Você acertou em ${tentativas} tentativas!`);
        } else if (palpite < numeroSecreto) {
            alert("Tente um número MAIOR!");
        } else {
            alert("Tente um número MENOR!");
        }
    }
    
    if (!acertou) {
        alert(`Game Over! O número era ${numeroSecreto}`);
    }

    totalJogos++;
    document.getElementById('jogosJogados').textContent = totalJogos;
}
